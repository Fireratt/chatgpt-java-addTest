         System.out.print(prompt);
         BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
         List<String> lines = new ArrayList<>();
         String line;
         try {
             while ((line = reader.readLine()) != null && !line.isEmpty()) {
                 lines.add(line);
